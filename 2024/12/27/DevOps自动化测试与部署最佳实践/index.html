import datetime
from github import Github

def analyze_deployments(repo_name, token, days=30):
    g = Github(token)
    repo = g.get_repo(repo_name)
    
    # è·å–æ‰€æœ‰å·¥ä½œæµè¿è¡Œ
    workflows = repo.get_workflows()

    # åˆ†æéƒ¨ç½²
    cutoff_date = datetime.datetime.now() - datetime.timedelta(days=days)
    runs = deploy_workflow.get_runs()
    
    successful_deploys = 0
    failed_deploys = 0
    
    for run in runs:
        if run.created_at < cutoff_date:
            continue
            
        if run.conclusion == "success":
            successful_deploys += 1
        elif run.conclusion == "failure":
            failed_deploys += 1
    
    total_deploys = successful_deploys + failed_deploys
    success_rate = (successful_deploys / total_deploys * 100) if total_deploys > 0 else 0
    
    return {
        "total_deploys": total_deploys,
        "deploys_per_day": total_deploys / days,
        "success_rate": success_rate
    }
