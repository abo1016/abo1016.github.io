func NewDB(config *Config) (*sql.DB, error) {
    db, err := sql.Open("postgres", config.DBConnectionString)

        return nil, err

        return nil, err

    db *sql.DB

    return &UserRepository{db: db}
}

    // ä»æ•°æ®åº“è·å–ç”¨æˆ·
    return &User{ID: id, Name: "User " + id}, nil

type UserService struct {
    repo *UserRepository
}

func NewUserService(repo *UserRepository) *UserService {
    return &UserService{repo: repo}
}

func (s *UserService) GetUser(id string) (*User, error) {
    return s.repo.GetByID(id)
}

type UserHandler struct {
    service *UserService
}

func NewUserHandler(service *UserService) *UserHandler {
    return &UserHandler{service: service}
}

func (h *UserHandler) GetUser(w http.ResponseWriter, r *http.Request) {
    id := r.URL.Query().Get("id")
    if id == "" {
        http.Error(w, "Missing id parameter", http.StatusBadRequest)
        return
    }
    
    user, err := h.service.GetUser(id)
    if err != nil {
        http.Error(w, err.Error(), http.StatusInternalServerError)
        return
    }
    
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(http.StatusOK)
    w.Write([]byte(`{"id":"` + user.ID + `","name":"` + user.Name + `"}`))
}

type Server struct {
    handler *UserHandler
    port    string
}

func NewServer(handler *UserHandler, config *Config) *Server {
    return &Server{
        handler: handler,
        port:    config.ServerPort,
    }
}

func (s *Server) Start() error {
    http.HandleFunc("/user", s.handler.GetUser)
    return http.ListenAndServe(s.port, nil)
}

    server, err := InitializeServer()

    }
    
    log.Printf("Starting server on port %s", server.port)
    if err := server.Start(); err != nil {
        log.Fatalf("Server error: %v", err)
    }
}
