type ModbusAdapter struct {
    config     ModbusConfig
    client     modbus.Client
    deviceChan chan DeviceData
    ctx        context.Context
    cancel     context.CancelFunc
    logger     *zap.Logger
}

func NewModbusAdapter(config ModbusConfig) *ModbusAdapter {
    ctx, cancel := context.WithCancel(context.Background())
    return &ModbusAdapter{
        config:     config,
        deviceChan: make(chan DeviceData, 100),
        ctx:        ctx,
        cancel:     cancel,
        logger:     log.With(zap.String("adapter", "modbus")),
    }
}

func (a *ModbusAdapter) Connect() error {
    // æ ¹æ®é…ç½®åˆ›å»º Modbus å®¢æˆ·ç«¯
    handler := modbus.NewTCPClientHandler(fmt.Sprintf("%s:%d", a.config.Host, a.config.Port))
    handler.Timeout = time.Second * 5
    if err := handler.Connect(); err != nil {
        return errors.Wrap(err, "failed to connect modbus device")
    }
    
    a.client = modbus.NewClient(handler)
    a.startPolling()
    return nil
}

func (a *ModbusAdapter) startPolling() {
    go func() {
        ticker := time.NewTicker(time.Duration(a.config.PollInterval) * time.Millisecond)
        defer ticker.Stop()
        
        for {
            select {
            case <-ticker.C:
                for _, point := range a.config.Points {
                    value, err := a.readPoint(point)
                    if err != nil {
                        a.logger.Error("failed to read point", zap.Error(err))
                        continue
                    }
                    
                    a.deviceChan <- DeviceData{
                        DeviceID:  a.config.DeviceID,
                        Timestamp: time.Now(),
                        Point:     point.Name,
                        Value:     value,
                    }
                }
            case <-a.ctx.Done():
                return
            }
        }
    }()
}

func (a *ModbusAdapter) readPoint(point ModbusPoint) (interface{}, error) {
    var result []byte
    var err error
    
    switch point.FunctionCode {
    case 1: // Read Coils
        result, err = a.client.ReadCoils(point.Address, point.Quantity)
    case 2: // Read Discrete Inputs
        result, err = a.client.ReadDiscreteInputs(point.Address, point.Quantity)
    case 3: // Read Holding Registers
        result, err = a.client.ReadHoldingRegisters(point.Address, point.Quantity)
    case 4: // Read Input Registers
        result, err = a.client.ReadInputRegisters(point.Address, point.Quantity)
    default:
        return nil, errors.New("unsupported function code")
    }
    
    if err != nil {
        return nil, err
    }
    
    return a.convertValue(result, point.DataType), nil
}
